83  {
84  crc ^= data[i];

302 errorTypes Tle5012b::checkSafety(uint16_t safety, uint16_t command, uint16_t* readreg, uint16_t length)
303 {
304  errorTypes errorCheck;
305  safetyWord = safety;
306 
307  if (!((safety) & SYSTEM_ERROR_MASK))
308  {
309  errorCheck = SYSTEM_ERROR;
310  //resetSafety();
311  } else if (!((safety) & INTERFACE_ERROR_MASK))
312  {
313  errorCheck = INTERFACE_ERROR_MASK;
314  //resetSafety();
315  } else if (!((safety) & INV_ANGLE_ERROR_MASK))
316  {
317  errorCheck = INV_ANGLE_ERROR_MASK;
318  //resetSafety();
319  }else{
320  //resetSafety();
321  uint16_t lengthOfTemp = length * 2 + 2;
322  uint8_t temp[lengthOfTemp];
323 
324  temp[0] = getFirstByte(command);
325  temp[1] = getSecondByte(command);
326 
327  for (uint16_t i = 0; i < length; i++)
328  {
329  temp[2 + 2 * i] = getFirstByte(readreg[i]);
330  temp[2 + 2 * i + 1] = getSecondByte(readreg[i]);
331  }
332 
333  uint8_t crcReceivedFinal = getSecondByte(safety);
334  uint8_t crc = crcCalc(temp, lengthOfTemp);
335 
336  if (crc == crcReceivedFinal)
337  {
338  errorCheck = NO_ERROR;
339  }else{
340  errorCheck = CRC_ERROR;
341  resetSafety();
342  }
343  }
344  return (errorCheck);

355 errorTypes Tle5012b::regularCrcUpdate()
356 {
357  readBlockCRC();
358  uint8_t temp[16];
359  for (uint8_t i = 0; i < CRC_NUM_REGISTERS; i++)
360  {
361  temp[2 * i] = getFirstByte(_registers[i]);
362  temp[(2 * i) + 1] = getSecondByte(_registers[i]);
363  }
364  uint8_t crc = crcCalc(temp, 15);
365  uint16_t firstTempByte = (uint16_t) temp[14];
366  uint16_t secondTempByte = (uint16_t) crc;
367  uint16_t valToSend = (firstTempByte << 8) | secondTempByte;
368  _registers[7] = valToSend;
369 
370  return (writeTempCoeffUpdate(valToSend));
